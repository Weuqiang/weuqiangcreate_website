import re

# findall
target = 'life is short, i learn python.'
result = re.findall('python', target)
result1 = re.findall('java', target)
# findallrefindall
# resulttarget'python',
print(result)# ['python']

target = 'abc acc aec agc adc aic'
result = re.findall('a[de]c', target)
# [de]de
print(result)# ['aec', 'adc']

result = re.findall('a[b‐z]c', target)
# [b‐z]b‐z
print(result)# ['abc', 'acc', 'aec', 'agc', 'adc', 'aic']

result = re.findall('a[^c‐z]c', target)
# [^c‐z]c‐z
print(result)# ['abc']

line = "Cats are smarter than dogs"
matchObj = re.match( r'(.*) are (.*?) .*', line, re.M|re.I)
#re.M ^  $
#re.I 
if matchObj:
   print ("matchObj.group() : ", matchObj.group())#
   print ("matchObj.group(1) : ", matchObj.group(1)) # 10
   print ("matchObj.group(2) : ", matchObj.groups())# 
else:
   print ("No match!!")

def double(matched):
    value = int(matched.group('value'))
    return str(value * 2)
s = 'A23G4HFD567'
print(re.sub('(?P<value>\d+)', double, s))
