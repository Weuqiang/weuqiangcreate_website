}

#define PRINT_ARRAY(arr, size) \
    do { \
        printf("["); \

            if (i < (size) - 1) { \

#include <stdio.h>

#define BENCHMARK(name, code) \
    do { \
        clock_t start = clock(); \
        code; \
        clock_t end = clock(); \
        double time_spent = (double)(end - start) / CLOCKS_PER_SEC; \
        printf("%s: %.6f seconds\n", name, time_spent); \
    } while(0)

int main() {
    BENCHMARK("Loop test", {
        int sum = 0;
        for (int i = 0; i < 1000000; i++) {
            sum += i;
        }
    });

        int arr[1000];

            arr[i] = i * 2;
        }
    });
    
    return 0;
}
