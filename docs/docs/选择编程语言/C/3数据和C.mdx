#### char

```c showLineNumbers
#include <stdio.h>

int main() {
    char ch1 = 'A';        // 
    char ch2 = 65;         // ASCII
    char ch3 = '\n';       // 
    
    printf("ch1 = %c (ASCII: %d)\n", ch1, ch1);
    printf("ch2 = %c (ASCII: %d)\n", ch2, ch2);
    
    // char
    char a = 'A';
    char b = a + 1;
    printf("'A' + 1 = %c\n", b);  // 'B'
    
    // 
    char upper = 'A';
    char lower = upper + 32;  // 
    printf("%c -> %c\n", upper, lower);
    
    return 0;
}
```

### 

|  |  |  |
|--------|---------|------|
| `%d` / `%i` |  | `printf("%d", 123);` |
| `%u` |  | `printf("%u", 123);` |
| `%o` |  | `printf("%o", 123);` |
| `%x` / `%X` |  | `printf("%x", 123);` |
| `%f` |  | `printf("%f", 3.14);` |
| `%e` / `%E` |  | `printf("%e", 3.14);` |
| `%g` / `%G` | %f%e | `printf("%g", 3.14);` |
| `%c` |  | `printf("%c", 'A');` |
| `%s` |  | `printf("%s", "Hello");` |
| `%p` |  | `printf("%p", &var);` |
| `%%` |  | `printf("%%");` |

### 

```c showLineNumbers
#include <stdio.h>

int main() {
    int num = 123;
    float pi = 3.14159;
    
    // 
    printf("%5d\n", num);      // "  123" (5)
    printf("%-5d\n", num);     // "123  " (5)
    printf("%05d\n", num);     // "00123" (0)
    
    // 
    printf("%.2f\n", pi);      // "3.14" (2)
    printf("%.4f\n", pi);      // "3.1416" (4)
    printf("%8.2f\n", pi);     // "    3.14" (82)
    
    // 
    long l = 123456789L;
    long long ll = 123456789012345LL;
    printf("%ld\n", l);        // long
    printf("%lld\n", ll);      // long long
    
    // 
    printf("%#x\n", 255);      // "0xff" ()
    printf("%+d\n", 123);      // "+123" ()
    printf("% d\n", 123);      // " 123" ()
    
    return 0;
}
```

scanf&

### 

```c showLineNumbers
#include <stdio.h>

int main() {
    int num;
    float price;
    char ch;
    char name[50];
    
    // 
    printf("Enter a number: ");
    scanf("%d", &num);
    
    // 
    printf("Enter a price: ");
    scanf("%f", &price);
    
    // 
    printf("Enter a character: ");
    scanf(" %c", &ch);  // 
    
    // 
    printf("Enter your name: ");
    scanf("%s", name);  // &
    
    printf("\nYou entered:\n");
    printf("Number: %d\n", num);
    printf("Price: %.2f\n", price);
    printf("Character: %c\n", ch);
    printf("Name: %s\n", name);
    
    return 0;
}
```
