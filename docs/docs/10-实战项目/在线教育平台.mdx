@Service
@Slf4j
public class CourseService {
    
    @Autowired
    private CourseMapper courseMapper;

    @Autowired
    private RedisTemplate<String, Object> redisTemplate;
    
    /**
     * 
     */
    public PageResult<CourseVO> getCourseList(CourseQueryDTO query) {
        Page<Course> page = new Page<>(query.getPage(), query.getPageSize());
        
        LambdaQueryWrapper<Course> wrapper = new LambdaQueryWrapper<>();
        wrapper.eq(Course::getStatus, CourseStatus.PUBLISHED)
               .like(StringUtils.hasText(query.getKeyword()), 
                     Course::getTitle, query.getKeyword())
               .eq(query.getCategoryId() != null, 
                   Course::getCategoryId, query.getCategoryId())
               .eq(query.getLevel() != null, 
                   Course::getLevel, query.getLevel())
               .orderByDesc(Course::getCreatedAt);
        
        Page<Course> result = courseMapper.selectPage(page, wrapper);
        
        List<CourseVO> voList = result.getRecords().stream()
            .map(this::convertToVO)
            .collect(Collectors.toList());
        
        return PageResult.of(voList, result.getTotal());
    }
    
    /**
     * 
     */
    public CourseDetailVO getCourseDetail(Long courseId) {
        // 
        String cacheKey = "course:detail:" + courseId;
        CourseDetailVO cached = (CourseDetailVO) redisTemplate.opsForValue().get(cacheKey);
        if (cached != null) {
            return cached;
        }
        
        // 
        Course course = courseMapper.selectById(courseId);
        if (course == null) {
            throw new BusinessException("");
        }
        
        // 
        List<Chapter> chapters = chapterMapper.selectList(
            new LambdaQueryWrapper<Chapter>()
                .eq(Chapter::getCourseId, courseId)
                .orderByAsc(Chapter::getSortOrder)
        );
        
        CourseDetailVO vo = convertToDetailVO(course, chapters);
        
        // 1
        redisTemplate.opsForValue().set(cacheKey, vo, 1, TimeUnit.HOURS);
        
        return vo;
    }
    
    /**
     * 
     */
    @Transactional
    public Long createCourse(CreateCourseDTO dto, Long teacherId) {
        Course course = new Course();
        BeanUtils.copyProperties(dto, course);
        course.setTeacherId(teacherId);
        course.setStatus(CourseStatus.DRAFT);
        
        courseMapper.insert(course);
        
        return course.getId();
    }

     * 
     */

        }

            throw new BusinessException("");
        }

        // 
        redisTemplate.delete("course:detail:" + courseId);

// LearningProgressService.java
@Service
public class LearningProgressService {
    
    @Autowired
    private LearningProgressMapper progressMapper;
    
    @Autowired
    private LessonMapper lessonMapper;
    
    /**
     * 
     */
    @Transactional
    public void updateProgress(Long userId, Long lessonId, 
                               Integer position, Integer duration) {
        LearningProgress progress = progressMapper.selectOne(
            new LambdaQueryWrapper<LearningProgress>()
                .eq(LearningProgress::getUserId, userId)
                .eq(LearningProgress::getLessonId, lessonId)
        );
        
        if (progress == null) {
            progress = new LearningProgress();
            progress.setUserId(userId);
            progress.setLessonId(lessonId);
        }
        
        progress.setLastPosition(position);
        
        // 
        int progressPercent = (int) ((position * 100.0) / duration);
        progress.setProgress(Math.min(progressPercent, 100));
        
        // 
        if (progressPercent >= 90) {
            progress.setIsCompleted(1);
        }
        
        if (progress.getId() == null) {
            progressMapper.insert(progress);
        } else {
            progressMapper.updateById(progress);
        }
    }
    
    /**
     * 
     */
    public CourseProgressVO getCourseProgress(Long userId, Long courseId) {
        // 
        List<Lesson> lessons = lessonMapper.selectByCourseId(courseId);
        
        // 
        List<LearningProgress> progressList = progressMapper.selectList(
            new LambdaQueryWrapper<LearningProgress>()
                .eq(LearningProgress::getUserId, userId)
                .in(LearningProgress::getLessonId, 
                    lessons.stream().map(Lesson::getId).collect(Collectors.toList()))
        );
        
        Map<Long, LearningProgress> progressMap = progressList.stream()
            .collect(Collectors.toMap(LearningProgress::getLessonId, p -> p));
        
        // 
        int totalLessons = lessons.size();
        int completedLessons = (int) progressList.stream()
            .filter(p -> p.getIsCompleted() == 1)
            .count();
        
        int overallProgress = totalLessons > 0 
            ? (completedLessons * 100 / totalLessons) 
            : 0;
        
        CourseProgressVO vo = new CourseProgressVO();
        vo.setCourseId(courseId);
        vo.setTotalLessons(totalLessons);
        vo.setCompletedLessons(completedLessons);
        vo.setOverallProgress(overallProgress);
        vo.setLessonProgress(progressMap);
        
        return vo;
    }
}

// FileUploadService.java
@Service
public class FileUploadService {
    
    @Autowired
    private MinioClient minioClient;
    
    @Value("${minio.bucket}")
    private String bucket;
    
    /**
     * 
     */
    public String uploadVideo(MultipartFile file) {
        try {
            String fileName = generateFileName(file.getOriginalFilename());
            String objectName = "videos/" + fileName;
            
            minioClient.putObject(
                PutObjectArgs.builder()
                    .bucket(bucket)
                    .object(objectName)
                    .stream(file.getInputStream(), file.getSize(), -1)
                    .contentType(file.getContentType())
                    .build()
            );
            
            return getFileUrl(objectName);
        } catch (Exception e) {
            throw new BusinessException("");
        }
    }
    
    /**
     * 
     */
    private String generateFileName(String originalFilename) {
        String extension = originalFilename.substring(
            originalFilename.lastIndexOf(".")
        );
        return UUID.randomUUID().toString() + extension;
    }
    
    /**
     * URL
     */
    private String getFileUrl(String objectName) {
        try {
            return minioClient.getPresignedObjectUrl(
                GetPresignedObjectUrlArgs.builder()
                    .bucket(bucket)
                    .object(objectName)
                    .expiry(7, TimeUnit.DAYS)
                    .build()
            );
        } catch (Exception e) {
            throw new BusinessException("URL");
        }
    }
}

import { useEffect, useRef } from 'react';
import videojs from 'video.js';
import 'video.js/dist/video-js.css';

interface VideoPlayerProps {
  src: string;
  onTimeUpdate?: (currentTime: number) => void;
  onEnded?: () => void;
  initialTime?: number;
}

export default function VideoPlayer({
  src,
  onTimeUpdate,
  onEnded,
  initialTime = 0,
}: VideoPlayerProps) {
  const videoRef = useRef<HTMLVideoElement>(null);
  const playerRef = useRef<any>(null);
  
  useEffect(() => {
    if (!videoRef.current) return;
    
    // 
    const player = videojs(videoRef.current, {
      controls: true,
      autoplay: false,
      preload: 'auto',
      fluid: true,
      sources: [{
        src,
        type: 'video/mp4',
      }],
    });
    
    playerRef.current = player;
    
    // 
    if (initialTime > 0) {
      player.currentTime(initialTime);
    }
    
    // 
    player.on('timeupdate', () => {
      const currentTime = player.currentTime();
      onTimeUpdate?.(currentTime);
    });
    
    // 
    player.on('ended', () => {
      onEnded?.();
    });
    
    return () => {
      if (playerRef.current) {
        playerRef.current.dispose();
      }
    };
  }, [src]);
  
  return (
    <div data-vjs-player>
      <video ref={videoRef} className="video-js" />
    </div>
  );
}
