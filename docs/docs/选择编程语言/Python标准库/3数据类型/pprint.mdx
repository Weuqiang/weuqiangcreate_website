## pprint 

`pprint` Python

### 

```python showLineNumbers
import pprint

data = {
    'name': 'Python',
    'config': {
        'debug': True,
        'database': {
            'host': 'localhost',
            'credentials': {'user': 'admin', 'password': '***'}
        }
    },
    'users': [
        {'id': 1, 'name': '', 'email': 'zhangsan@example.com'},
        {'id': 2, 'name': '', 'email': 'lisi@example.com'},
    ]
}

# print
print(data)

# pprint.pp
pprint.pp(data)
```

### 

```python showLineNumbers
import pprint

data = {'z': 1, 'a': 2, 'b': [1, 2, 3], 'c': {'d': 4}}

# pp() - 
pprint.pp(data, 
    width=40,        # 40
    indent=4,        # 4
    depth=2,         # 2
    compact=True,    # 
    sort_dicts=False # 
)

# pformat() - 
formatted = pprint.pformat(data, indent=4)
with open('output.txt', 'w') as f:
    f.write(formatted)

# pprint() - 
pprint.pprint(data)  # : {'a': 2, 'b': [...], 'c': {...}, 'z': 1}
```

### 

```python showLineNumbers
import pprint

# 
recursive_list = [1, 2, 3]
recursive_list.append(recursive_list)

pprint.pp(recursive_list)
# : [1, 2, 3, <Recursion on list with id=...>]

# /
print(pprint.isreadable(recursive_list))  # False
print(pprint.isrecursive(recursive_list))  # True
```

```python showLineNumbers
import logging
import pprint

logging.basicConfig(level=logging.DEBUG)

def process_order(order):
    logging.debug(f":\n{pprint.pformat(order, indent=4)}")
    
order = {
    'id': '12345',
    'items': [{'name': 'A', 'price': 99.9, 'qty': 2}],
    'customer': {'name': '', 'address': ''}
}
process_order(order)
```

### PrettyPrinter



```python showLineNumbers
import pprint

pp = pprint.PrettyPrinter(indent=4, width=60, depth=3, compact=True)

data1 = {'name': 'Alice', 'scores': [95, 87, 92]}
data2 = {'name': 'Bob', 'scores': [88, 90, 85]}

pp.pprint(data1)
pp.pprint(data2)
```
