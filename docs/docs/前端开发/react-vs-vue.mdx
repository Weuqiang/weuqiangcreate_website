#### 

**Vue**
```vue
<template>
  <div>
    <h1>{{ title }}</h1>
    <button @click="increment">{{ count }}</button>
  </div>
</template>

<script>
export default {
  data() {
    return {
      title: 'Vue',
      count: 0
    }
  },
  methods: {
    increment() {
      this.count++
    }
  }
}
</script>
```

**React**
```jsx
import React, { useState } from 'react'

function App() {
  const [count, setCount] = useState(0)
  const title = 'React'
  
  const increment = () => {
    setCount(count + 1)
  }
  
  return (
    <div>
      <h1>{title}</h1>
      <button onClick={increment}>{count}</button>
    </div>
  )
}
```
