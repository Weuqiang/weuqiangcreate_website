from sklearn.decomposition import PCA
import numpy as np

# 
X = np.array([[1, 2, 3],
              [2, 3, 4],
              [3, 4, 5],
              [4, 5, 6]])  # 

# PCA
n_components = 2  # 
model = PCA(n_components=n_components)

# .fit_transform()
X_reduced = model.fit_transform(X)

print(":", X.shape)
print(":", X_reduced.shape)
print(":")
print(X_reduced)
