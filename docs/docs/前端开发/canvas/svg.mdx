### Path2D

CanvasAPI`Path2D`svg`d`Canvas

``` js
const ctx = canvas.getContext('2d');
const svgPath = `M10 10 L10 50 L 50 50 Z`;
const path = new Path2D(svgPath);
ctx.strokeStyle = 'pink'
ctx.stroke(path)
```

``` js
const ctx = canvas.getContext('2d');
const svgPath = `M10 10 L10 50 L 50 50 Z`;
const path = new Path2D(svgPath);
ctx.fillStyle = 'pink'
ctx.fill(path)
```

``` js
const ctx = canvas.getContext('2d');
const svgPath = `M10 10 L10 50 L 50 50 Z`;
const path = new Path2D(svgPath);
ctx.clip(path)

ctx.fillStyle = 'pink'
ctx.fillRect(0, 0, 30, 100)
```
